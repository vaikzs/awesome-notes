    s->elements = (int *)malloc(s->allocLen * elemSize);

    // NULLæ£€æŸ¥

void StackPush(stack *s, void *elemAddr) {
    // æ»¡æ ˆæ£€æŸ¥
    if (s->logLen == s->allocLen) {

    void *target = (char *)s->elements + s->logLen * s->elemSize;
    memcpy(target, elemAddr, s->elemSize);
    s->logLen++;
}
